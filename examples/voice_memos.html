from superduperdb.datalayer.mongodb.client import SuperDuperClient
from superduperdb.datalayer.mongodb.query import Insert

import glob

c = SuperDuperClient()
c.drop_database('voicememos', force=True)
db = c.voicememos

memos = glob.glob('data/voice_memos/*.wav')
data = [
    {'audio': {'_content': {'uri': f'file://{memo}', 'type': 'silero'}}}
    for memo in memos
]

db.insert(Insert(documents=data, collection='documents'))
